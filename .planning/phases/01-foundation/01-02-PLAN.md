// Sentinel-based localStorage availability check.
// Called once at app startup in App.tsx.
// Source: MDN Window.localStorage — https://developer.mozilla.org/en-US/docs/Web/API/Window/localStorage
// Source: WebKit Blog storage policy — https://webkit.org/blog/14403/updates-to-storage-policy/

export function isStorageAvailable(): boolean {
  try {
    const testKey = '__lift_calc_test__'
    localStorage.setItem(testKey, '1')
    localStorage.removeItem(testKey)
    return true
  } catch {
    return false
  }
}

// All AppState types. The store shape is the contract for localStorage serialization.
// CRITICAL: schemaVersion must be present from the first write — changing the shape
// without bumping schemaVersion causes silent data corruption on existing installs.

export interface Exercise {
  id: string
  name: string
  isCustom: boolean
}

export interface AppState {
  schemaVersion: number          // Always present. Current version: 1.
  exercises: Exercise[]          // Populated with seed data in Phase 2.
  maxWeights: Record<string, number>  // exerciseId -> 1RM in kg. Set in Phase 2.
}

export interface AppStore extends AppState {
  // Actions added in Phase 2 and 3.
  setMaxWeight: (exerciseId: string, weight: number) => void
}

// src/store/store.ts
// Single Zustand store for all app state.
// Uses `persist` middleware to write full store to localStorage under key `lift-calc-v1`.
// Source: https://zustand.docs.pmnd.rs/reference/middlewares/persist

import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import type { AppStore } from './types'

const CURRENT_SCHEMA_VERSION = 1

export const useAppStore = create<AppStore>()(
  persist(
    (set) => ({
      // Foundation fields — present from first write, required for migrations.
      schemaVersion: CURRENT_SCHEMA_VERSION,

      // Phase 2 will populate these with seed data.
      exercises: [],
      maxWeights: {},

      // Actions — Phase 2 will add more.
      setMaxWeight: (exerciseId, weight) =>
        set((state) => ({
          maxWeights: { ...state.maxWeights, [exerciseId]: weight },
        })),
    }),
    {
      name: 'lift-calc-v1',  // localStorage key. Versioned so future breaking changes can use a new key.
      onRehydrateStorage: () => (_state, error) => {
        if (error) {
          // Fires if localStorage is available but data is corrupted (e.g., malformed JSON).
          // The storage-unavailable case is caught earlier by isStorageAvailable() in App.tsx.
          console.warn('[lift-calc] Storage rehydration failed:', error)
        }
      },
    }
  )
)
```

// src/App.tsx
// Storage gate: if localStorage is unavailable (Safari private browsing),
// render a graceful warning instead of the app. Prevents QuotaExceededError crashes.
import { useState } from 'react'
import { isStorageAvailable } from './lib/storage'

function StorageWarning() {
  return (
    <div className="flex items-center justify-center min-h-[100dvh] p-6 text-center bg-gray-950 text-white">
      <div className="max-w-sm">
        <p className="text-lg font-semibold mb-2">Storage unavailable</p>
        <p className="text-sm text-gray-400">
          You're in private browsing mode. Lift Calc needs localStorage to save
          your data. Open in a regular browser window to use the app.
        </p>
      </div>
    </div>
  )
}

export default function App() {
  // useState initializer runs once — checks localStorage availability at mount.
  // Using an initializer (not useEffect) means the gate renders synchronously,
  // so the app never partially renders before showing the warning.
  const [storageAvailable] = useState(() => isStorageAvailable())

  if (!storageAvailable) {
    return <StorageWarning />
  }

  return (
    <div className="min-h-[100dvh] flex flex-col bg-gray-950 text-white">
      <main className="flex-1 p-4">
        {/* Phase 2 renders exercise list here */}
        <p className="text-gray-500 text-sm">Foundation ready. Phase 2 content loads here.</p>
      </main>
    </div>
  )
}
```
